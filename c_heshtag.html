public class Auto {
    private string marka;
    public string Marka {
        get { return marka; }
        set { marka = value; }
    }

    public int Evjarat { get; set; }

    public Auto(string marka, int evjarat) {
        this.marka = marka;
        this.Evjarat = evjarat;
    }

    public void Indit() {

Ciklusok

A C#-ban többféle ciklus használható:


// For ciklus
for (int i = 0; i < 5; i++) {
    Console.WriteLine(i);
}

// While ciklus
int j = 0;
while (j < 5) {
    Console.WriteLine(j);
    j++;
}

// Foreach ciklus
string[] nevek = {"Anna", "Béla", "Cecil"};
foreach (string nev in nevek) {
    Console.WriteLine(nev);
}
